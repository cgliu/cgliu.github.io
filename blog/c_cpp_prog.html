class MemoryBlock
{
public:
    // Simple constructor that initializes the resource.
    explicit MemoryBlock(size_t length)
        : _length(length)
        , _data(new int[length])
    {
        std::cout << "In MemoryBlock(size_t). length = "
                  << _length << "." << std::endl;
    }

    // Destructor.
    ~MemoryBlock()
    {
        std::cout << "In ~MemoryBlock(). length = "
                  << _length << ".";
        if (_data != nullptr)
        {
            std::cout << " Deleting resource.";
            // Delete the resource.
            delete[] _data;
        }
        std::cout << std::endl;
    }

    // Copy constructor.
    MemoryBlock(const MemoryBlock& other)
        : _length(other._length)
        , _data(new int[other._length])
    {
        std::cout << "In MemoryBlock(const MemoryBlock&). length = "
                  << other._length << ". Copying resource." << std::endl;

        std::copy(other._data, other._data + _length, _data);
    }

    // Copy assignment operator.
    MemoryBlock& operator=(const MemoryBlock& other)
    {
        std::cout << "In operator=(const MemoryBlock&). length = "
                  << other._length << ". Copying resource." << std::endl;

        if (this != &other)
        {
            // Free the existing resource.
            delete[] _data;

            _length = other._length;
            _data = new int[_length];
            std::copy(other._data, other._data + _length, _data);
        }
        return *this;
    }

    // Retrieves the length of the data resource.
    size_t Length() const
    {
        return _length;
    }

    // Move constructor.
    MemoryBlock(MemoryBlock&& other)
        : _data(nullptr)
        , _length(0)
    {
        std::cout << "In MemoryBlock(MemoryBlock&&). length = "
                  << other._length << ". Moving resource." << std::endl;

        // Copy the data pointer and its length from the
        // source object.
        _data = other._data;
        _length = other._length;

        // Release the data pointer from the source object so that
        // the destructor does not free the memory multiple times.
        other._data = nullptr;
        other._length = 0;
    }

    // Move assignment operator.
    MemoryBlock& operator=(MemoryBlock&& other)
    {
        std::cout << "In operator=(MemoryBlock&&). length = "
                  << other._length << "." << std::endl;

        if (this != &other)
        {
            // Free the existing resource.
            delete[] _data;

            // Copy the data pointer and its length from the
            // source object.
            _data = other._data;
            _length = other._length;

            // Release the data pointer from the source object so that
            // the destructor does not free the memory multiple times.
            other._data = nullptr;
            other._length = 0;
        }
        return *this;
    }
private:
    size_t _length; // The length of the resource.
    int* _data; // The resource.
};

class HasPtr{
public:
    // Initialization
    HasPtr(const std::string &s=std::string()) :
        ps(new std::string(s)), i(0) {}
    // copy-constructor
    HasPtr(const HasPtr & p) :
        ps(new std::string(*p.ps)), i(p.i) {}
        //
    HasPtr & operator=(const HasPtr & rhs) {
        // value-like copy-assignment operation
        auto newps = new std::string ( *rhs.ps);
        delete ps;
        ps = newps;
        i = rhs.i;
        return * this;
    }
    ~HasPtr() { delete ps;}
private:
    std::string *ps;
    int i;
};







Here is an example of a class that acts like pointers





class HasPtr{
public:
    // Initialization
    HasPtr(const std::string &s=std::string()) :
        ps(new std::string(s)), i(0), use(new std::size_t(1) {}

    HasPtr(const HasPtr & p) :
        ps(*p.ps), i(p.i), use(p.use) { ++(*use);}

    HasPtr & operator=(const HasPtr & rhs) {
        ++ (*rhs.use);
        if(--(*use) == 0){
            delete use;
            delete ps;
        }
        ps = rhs.ps;
        i = rhs.i;
        use = rhs.use;
        return * this;
    }
    ~HasPtr() {
        --(*use);
        if(*use == 0){
            delete use;
            delete ps;
        }
    }
private:
    std::string *ps;
    int i;
    std::size_t *use; // keep track of how many objects share *ps
};

// A dynamic programming solution for longest palindr.
// This code is adopted from following link
// http://www.leetcode.com/2011/11/longest-palindromic-substring-part-i.html

#include <stdio.h>
#include <string.h>

// A utility function to print a substring str[low..high]
void printSubStr( char* str, int low, int high )
{
    for( int i = low; i <= high; ++i )
        printf("%c", str[i]);
}

// This function prints the longest palindrome substring
// of str[].
// It also returns the length of the longest palindrome
int longestPalSubstr( char *str )
{
    int n = strlen( str ); // get length of input string

    // table[i][j] will be false if substring str[i..j]
    // is not palindrome.
    // Else table[i][j] will be true
    bool table[n][n];
    memset(table, 0, sizeof(table));

    // All substrings of length 1 are palindromes
    int maxLength = 1;
    for (int i = 0; i < n; ++i)
        table[i][i] = true;

    // check for sub-string of length 2.
    int start = 0;
    for (int i = 0; i < n-1; ++i)
    {
        if (str[i] == str[i+1])
        {
            table[i][i+1] = true;
            start = i;
            maxLength = 2;
        }
    }

    // Check for lengths greater than 2. k is length
    // of substring
    for (int k = 3; k <= n; ++k)
    {
        // Fix the starting index
        for (int i = 0; i < n-k+1 ; ++i)
        {
            // Get the ending index of substring from
            // starting index i and length k
            int j = i + k - 1;

            // checking for sub-string from ith index to
            // jth index iff str[i+1] to str[j-1] is a
            // palindrome
            if (table[i+1][j-1] && str[i] == str[j])
            {
                table[i][j] = true;

                if (k > maxLength)
                {
                    start = i;
                    maxLength = k;
                }
            }
        }
    }

    printf("Longest palindrome substring is: ");
    printSubStr( str, start, start + maxLength - 1 );

    return maxLength; // return length of LPS
}

    STATES ret;

        {
            if(row.at(i) == '.')
                continue;
            if(row.at(i) != s.at(i))

///
/// Find the min distance from vertex i, passing all vertices, and return back to vertex 0
///
tuple<double, int> find_min_in_subset(const int i, const vector<int> & vertices, const matrix_d & W, const matrix_d & D)
{
    double min_dist = numeric_limits<double>::max();
    int min_index = -1;
    const int num_vertices = W.size();
    int key = convert_to_key(vertices); // covert a set of indices to a key
    for(int j : vertices)
    {
        int key_minus_j = key ^ (1 << j); // remove j from subset
        double  dist = W[i][j] + D[j][key_minus_j];
        if(dist < min_dist)
        {
            min_dist = dist;
            min_index = j;
        }
    }

}

///
/// Travel through all vertices for once. 
///
/// @return min distance matrix and its corresponding index matrix
///
tuple<matrix_d, matrix_i> travel(const matrix_d & W)
{
    const int num_vertices = W.size();
    const int num_subsets = pow(2, num_vertices);

    matrix_d D(num_vertices, vector<double>( num_subsets, 0));

    matrix_i P(num_vertices, vector<int>(num_subsets, 0));

    // Indecies from 1 to N -1
    vector<int> other_vertices(num_vertices - 1);
    iota(other_vertices.begin(), other_vertices.end(), 1);

    // Vertex 0 is the starting point
    const int empty_set_key = convert_to_key({});
    for(int i = 1; i < num_vertices; ++i)
    {
        D[i][empty_set_key] = W[i][0];
        P[i][empty_set_key] = 0;
    }

    for(int k = 1; k <= num_vertices - 2; ++k)
    {
        for(auto ksubset: get_ksubset(k, other_vertices))
        {
            const int subset_key = convert_to_key(ksubset);
            for(int i = 1; i < num_vertices; ++i)
            {
                if(find(ksubset.begin(), ksubset.end(), i) == ksubset.end())
                {
                    tie(D[i][subset_key], P[i][subset_key]) = find_min_in_subset(i, ksubset, W, D);
                }
            }
        }
    }
    {
        const int key = convert_to_key(other_vertices); 
        tie(D[0][key], P[0][key]) = find_min_in_subset(0, other_vertices, W, D);
    }

    return make_tuple(D, P);
}

double length(const node & n, const matrix_d & W)
{
    double totlength = 0.0;
    for(int i = 0; i < n.path.size() - 1; ++i)
    {
        int curr_index = n.path[i];
        int next_index = n.path[i+1];

double min_length_to_go(const vector<int> & vertices, const matrix_d & W)
{
    double length = 0.0;
    for(int i = 0; i < vertices.size(); ++i)
    {
        double min_length = numeric_limits<double>::infinity();
        for(int k = 0; k < vertices.size(); ++k)
        {
            min_length = min(W[vertices[i]][vertices[k]], min_length); 
        }
        length += min_length;
    }

    return length;
}

double bound(const node & n, const matrix_d & W)
{
    const auto & path = n.path;
    double length_to_come = length(n, W);

    // find the min distance of path.back()
    vector<int> rest = {path.back()};
    for(int i = 1; i < W.size(); ++i)
    {
        if(in_path(i, path))
            continue;
        rest.push_back(i);
    }

    return length_to_come + min_length_to_go(rest, W);
}

vector<int> travel(const int num_vertices, const matrix_d & W,
                   const vector<array<double, 2>> & coordinates)
{
    vector<int> opt_path;
    auto cmp = [](const node & lhs, const node & rhs){
        return lhs.bound > rhs.bound;};
    priority_queue<node, vector<node>, decltype(cmp)> pq(cmp);

    node v;
    v.level = 0;
    v.path = {0};
    v.bound = bound(v, W);
    double min_length = numeric_limits<double>::infinity();
    pq.push(v);
    // return opt_path;

    while(!pq.empty())
    {
        // print_queue(pq);
        v = pq.top(); 
        pq.pop();
        if(v.bound < min_length)
        {
            node u;
            u.level = v.level + 1;
            for(int i = 1; i < num_vertices; ++i)
            {
                if(in_path(i, v.path))
                    continue;
                u.path = v.path;
                u.path.push_back(i);
                if(u.level == num_vertices - 2)
                {
                    insert_rest(u.path, num_vertices);
                    const auto & path_length = length(u, W);
                    if(path_length < min_length)
                    {
                        min_length = path_length;
                        opt_path = u.path;
                    }
                }
                else
                {
                    u.bound = bound(u, W);
                    if(u.bound < min_length)
                        pq.push(u);
                }
            }
        }
    }

    return opt_path;
}

double get_diff(double (*f)(double), double x)
{
  auto delta = 1e-2;
  auto f_plus = f(x + delta);
  auto f_minus = f(x - delta);
  return (f_plus - f_minus) / 2.0 / delta;
}

double fmin(double (*f)(double), double x0)
{
  auto f0 = f(x0);
  auto diff = get_diff(f, x0);
  auto x = x0;
  constexpr int max_iter = 5;
  while(abs(diff) > 1e-6)
    {
      auto alpha = 1.0;
      auto x = x0 - diff * alpha;
      if (f(x) < f0) // 
        {
          f0 = f(x);
          diff = get_diff(f, x);
          x0 = x;
        }
      else // line search
        {
          int iter = 1;
          while(iter < max_iter)
            {
              alpha = alpha * 0.5;
              x = x0 - diff * alpha;
              if (f(x) < f0) // amijo condition if second   
                {
                  f0 = f(x);
                  diff = get_diff(f, x);
                  x0 = x;
                  break;
                }
            }
          if(iter >= max_iter){
            throw runtime_error("failed to converge");
          }
        }
    }

  return f(x0);
}
